// Update horizon options based on period selection
function updateHorizonOptions() {
    const period = document.getElementById('periodSelect').value;
    const horizonSelect = document.getElementById('horizonSelect');
    
    // Clear existing options
    horizonSelect.innerHTML = '';
    
    let options = [];
    if (period === 'daily') {
        options = [
            { value: '7', text: '7 days' },
            { value: '14', text: '14 days' },
            { value: '21', text: '21 days' },
            { value: '30', text: '30 days' }
        ];
    } else if (period === 'weekly') {
        options = [
            { value: '4', text: '4 weeks' },
            { value: '8', text: '8 weeks' },
            { value: '12', text: '12 weeks' },
            { value: '16', text: '16 weeks' }
        ];
    } else if (period === 'monthly') {
        options = [
            { value: '3', text: '3 months' },
            { value: '6', text: '6 months' },
            { value: '12', text: '12 months' },
            { value: '24', text: '24 months' }
        ];
    }
    
    options.forEach(option => {
        const optionElement = document.createElement('option');
        optionElement.value = option.value;
        optionElement.textContent = option.text;
        horizonSelect.appendChild(optionElement);
    });
}

// Show/hide loading state
function showLoading(show) {
    const loadingState = document.getElementById('loadingState');
    const generateBtn = document.getElementById('generateBtn');
    
    if (show) {
        loadingState.style.display = 'block';
        generateBtn.disabled = true;
        generateBtn.innerHTML = '<i class="fas fa-spinner fa-spin me-2"></i>Generating...';
    } else {
        loadingState.style.display = 'none';
        generateBtn.disabled = false;
        generateBtn.innerHTML = '<i class="fas fa-chart-line me-2"></i>Generate Forecast';
    }
}

// Show error message
function showError(message) {
    const errorDisplay = document.getElementById('errorDisplay');
    errorDisplay.textContent = message;
    errorDisplay.style.display = 'block';
    
    // Hide error after 5 seconds
    setTimeout(() => {
        errorDisplay.style.display = 'none';

}

// Hide error message
function hideError() {
    const errorDisplay = document.getElementById('errorDisplay');
    errorDisplay.style.display = 'none';
}

// Generate forecast
async function generateForecast() {
    const medicineId = document.getElementById('medicineSelect').value;
    const period = document.getElementById('periodSelect').value;
    const horizon = document.getElementById('horizonSelect').value;
    
    if (!medicineId) {
        showError('Please select a medicine');
        return;
    }
    
    showLoading(true);
    hideError();
    
    try {
        const response = await fetch('{% url "analytics:api_generate_forecast_on_demand" %}', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-CSRFToken': document.querySelector('[name=csrfmiddlewaretoken]').value
            },
            body: JSON.stringify({
                medicine_id: medicineId,
                forecast_period: period,
                forecast_horizon: horizon
            })
        });
        
        const data = await response.json();
        
        if (response.ok) {
            // The API returns chart_data, so we need to pass the correct structure
            const forecastData = {
                medicine_name: data.medicine_name || 'Unknown Medicine',
                forecast_period: data.forecast_period || 'weekly',
                chart_data: data.chart_data || {},
                model_info: (data.chart_data && data.chart_data.model_info) ? data.chart_data.model_info : {}
            };
            displayForecast(forecastData);
        } else {
            if (data.error === 'insufficient_data') {
                showError(`${data.message} ${data.suggestion}`);
            } else {
                showError(`Error: ${data.error}`);
            }
        }
    } catch (error) {
        console.error('Error generating forecast:', error);
        showError('Failed to generate forecast. Please try again.');
    } finally {
        showLoading(false);
    }
}

// Create forecast chart
function createForecastChart(data) {
    // Destroy existing chart if it exists
    if (currentChart) {
        currentChart.destroy();
    }
    
    const canvas = document.getElementById('forecastChart');
    if (!canvas) {
        throw new Error('Canvas element not found');
    }
    
    const ctx = canvas.getContext('2d');
    
    // Validate data
    if (!data || (!data.historical && !data.forecast)) {
        throw new Error('Invalid chart data');
    }
    
    // Prepare data arrays
    const historicalData = (data.historical && data.historical.values) ? data.historical.values : [];
    const forecastData = (data.forecast && data.forecast.values) ? data.forecast.values : [];
    
    if (!Array.isArray(historicalData) || !Array.isArray(forecastData)) {
        throw new Error('Invalid data arrays');
    }
    
    // Create combined data arrays
    const combinedHistoricalData = [...historicalData];
    const combinedForecastData = [...Array(historicalData.length).fill(null), ...forecastData];
    
    // Create chart with zoom and pan functionality
    currentChart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: (data.labels && Array.isArray(data.labels)) ? data.labels : [],
            datasets: [{
                label: 'Historical Demand',
                data: combinedHistoricalData,
                borderColor: 'rgb(75, 192, 192)',
                backgroundColor: 'rgba(75, 192, 192, 0.2)',
                tension: 0.1,
                pointRadius: 4,
                pointHoverRadius: 8,
                pointHoverBorderWidth: 3,
                pointHoverBorderColor: 'rgb(75, 192, 192)',
                pointHoverBackgroundColor: 'white',
                fill: false,
                borderWidth: 2
            }, {
                label: 'Forecasted Demand',
                data: combinedForecastData,
                borderColor: 'rgb(255, 99, 132)',
                backgroundColor: 'rgba(255, 99, 132, 0.2)',
                tension: 0.1,
                pointRadius: 4,
                pointHoverRadius: 8,
                pointHoverBorderWidth: 3,
                pointHoverBorderColor: 'rgb(255, 99, 132)',
                pointHoverBackgroundColor: 'white',
                borderDash: [5, 5],
                fill: false,
                borderWidth: 2
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            animation: {
                duration: 1000,

            },
            plugins: {
                title: {
                    display: true,

                        size: 16,
                        weight: 'bold'
                    },
                    color: '#2c3e50'
                },
                legend: {
                    display: true,
                    position: 'top',
                    labels: {
                        usePointStyle: true,
                        pointStyle: 'circle',

                        font: {
                            size: 12
                        }
                    }
                },
                tooltip: {

                    mode: 'index',
                    intersect: false,
                    backgroundColor: 'rgba(0, 0, 0, 0.8)',
                    titleColor: 'white',
                    bodyColor: 'white',
                    borderColor: 'rgba(255, 255, 255, 0.2)',
                    borderWidth: 1,
                    cornerRadius: 8,
                    displayColors: true,
                    callbacks: {
                        title: function(tooltipItems) {
                            const labels = (data.labels && Array.isArray(data.labels)) ? data.labels : [];
                            return labels[tooltipItems[0].dataIndex] || 'Unknown Period';
                        },
                        label: function(context) {
                            let label = context.dataset.label || '';
                            if (label) {
                                label += ': ';
                            }
                            if (context.parsed.y !== null) {
                                label += context.parsed.y.toFixed(2) + ' units';
                            }
                            return label;
                        },
                        afterLabel: function(context) {
                            if (context.datasetIndex === 1 && context.parsed.y !== null) {
                                return 'Forecasted Value';
                            }
                            return '';
                        }
                    }
                },
                zoom: {
                    pan: {
                        enabled: false
                    },
                    zoom: {
                        wheel: {
                            enabled: true,

                        },

                        },
                        mode: 'xy'
                    }
                }
            },
            scales: {
                x: {

                        display: true,
                        text: 'Time Period',

                            size: 12,
                            weight: 'bold'
                        },

                        maxRotation: 45,

                            size: 10
                        },
                        color: '#6c757d'
                    },

                    }
                },

// Setup zoom controls
function setupZoomControls() {
    const zoomInBtn = document.getElementById('zoomInBtn');
    const zoomOutBtn = document.getElementById('zoomOutBtn');
    const resetZoomBtn = document.getElementById('resetZoomBtn');
    
    if (zoomInBtn) {
        zoomInBtn.addEventListener('click', function() {
            if (currentChart && currentChart.zoom) {
                currentChart.zoom.zoom(1.2);
            }
        });
    }
    
    if (zoomOutBtn) {
        zoomOutBtn.addEventListener('click', function() {
            if (currentChart && currentChart.zoom) {
                currentChart.zoom.zoom(0.8);
            }
        });
    }
    
    if (resetZoomBtn) {
        resetZoomBtn.addEventListener('click', function() {
            if (currentChart && currentChart.zoom) {
                currentChart.zoom.reset();
            }
        });
    }
}
